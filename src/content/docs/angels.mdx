// Angel visualization script
import * as d3 from 'https://cdn.skypack.dev/d3@7';

class AngelExplorer {
  constructor() {
    this.svg = d3.select('#angel-svg');
    this.width = 500;
    this.height = 500;
    
    this.wingSpan = 5;
    this.haloRadius = 2;
    this.frequency = 8;
    this.phase = 0;
    
    this.initializeControls();
    this.updateVisualization();
    this.startAnimation();
  }
  
  initializeControls() {
    document.getElementById('angel-wings').addEventListener('input', (e) => {
      this.wingSpan = parseFloat(e.target.value);
      document.getElementById('angel-wings-value').textContent = this.wingSpan.toFixed(1);
      this.updateVisualization();
    });
    
    document.getElementById('angel-halo').addEventListener('input', (e) => {
      this.haloRadius = parseFloat(e.target.value);
      document.getElementById('angel-halo-value').textContent = this.haloRadius.toFixed(1);
      this.updateVisualization();
    });
    
    document.getElementById('angel-frequency').addEventListener('input', (e) => {
      this.frequency = parseInt(e.target.value);
      document.getElementById('angel-frequency-value').textContent = this.frequency;
      this.updateVisualization();
    });
    
    document.getElementById('angel-phase').addEventListener('input', (e) => {
      this.phase = parseInt(e.target.value);
      document.getElementById('angel-phase-value').textContent = this.phase + 'Â°';
      this.updateVisualization();
    });
  }
  
  updateVisualization() {
    this.drawHalo();
    this.drawWings();
    this.drawBody();
    this.drawHarmonicPatterns();
  }
  
  drawHalo() {
    const haloGroup = this.svg.select('#angel-halo');
    haloGroup.selectAll('*').remove();
    
    // Main halo circle
    haloGroup.append('circle')
      .attr('cx', 0)
      .attr('cy', -6)
      .attr('r', this.haloRadius)
      .attr('fill', 'none')
      .attr('stroke', '#ffd700')
      .attr('stroke-width', 0.3)
      .attr('filter', 'url(#glow)')
      .attr('opacity', 0.8);
    
    // Inner glow
    haloGroup.append('circle')
      .attr('cx', 0)
      .attr('cy', -6)
      .attr('r', this.haloRadius * 0.7)
      .attr('fill', 'none')
      .attr('stroke', '#ffeb3b')
      .attr('stroke-width', 0.1)
      .attr('opacity', 0.6);
  }
  
  drawWings() {
    const wingsGroup = this.svg.select('#angel-wings');
    wingsGroup.selectAll('*').remove();
    
    const wingPath = d3.line()
      .x(d => d.x)
      .y(d => d.y)
      .curve(d3.curveBasis);
    
    // Left wing
    const leftWingData = this.generateWingData(-1);
    wingsGroup.append('path')
      .datum(leftWingData)
      .attr('d', wingPath)
      .attr('fill', 'rgba(255, 255, 255, 0.7)')
      .attr('stroke', '#e3f2fd')
      .attr('stroke-width', 0.2)
      .attr('filter', 'url(#glow)');
    
    // Right wing
    const rightWingData = this.generateWingData(1);
    wingsGroup.append('path')
      .datum(rightWingData)
      .attr('d', wingPath)
      .attr('fill', 'rgba(255, 255, 255, 0.7)')
      .attr('stroke', '#e3f2fd')
      .attr('stroke-width', 0.2)
      .attr('filter', 'url(#glow)');
  }
  
  generateWingData(side) {
    const data = [];
    const steps = 100;
    
    for (let i = 0; i <= steps; i++) {
      const t = (i / steps) * Math.PI * 2;
      const r = this.wingSpan * (1 + 0.5 * Math.cos(3 * t));
      const x = side * r * Math.cos(t) * 0.8;
      const y = r * Math.sin(t) * 0.4 - 2;
      data.push({ x, y });
    }
    
    return data;
  }
  
  drawBody() {
    const bodyGroup = this.svg.select('#angel-body');
    bodyGroup.selectAll('*').remove();
    
    // Head
    bodyGroup.append('circle')
      .attr('cx', 0)
      .attr('cy', -4)
      .attr('r', 1.2)
      .attr('fill', '#fff3e0')
      .attr('stroke', '#f5f5f5')
      .attr('stroke-width', 0.1);
    
    // Body
    bodyGroup.append('ellipse')
      .attr('cx', 0)
      .attr('cy', 0)
      .attr('rx', 1.5)
      .attr('ry', 3)
      .attr('fill', '#ffffff')
      .attr('stroke', '#f0f0f0')
      .attr('stroke-width', 0.1)
      .attr('opacity', 0.9);
  }
  
  drawHarmonicPatterns() {
    const patternGroup = this.svg.select('#harmonic-patterns');
    patternGroup.selectAll('*').remove();
    
    const phaseRad = (this.phase * Math.PI) / 180;
    
    for (let layer = 0; layer < 3; layer++) {
      const radius = 8 - layer * 2;
      const points = [];
      
      for (let angle = 0; angle < Math.PI * 2; angle += 0.1) {
        const r = radius + Math.sin(this.frequency * angle + phaseRad) * 0.5;
        const x = r * Math.cos(angle);
        const y = r * Math.sin(angle);
        points.push({ x, y });
      }
      
      const line = d3.line()
        .x(d => d.x)
        .y(d => d.y)
        .curve(d3.curveLinear);
      
      patternGroup.append('path')
        .datum(points)
        .attr('d', line)
        .attr('fill', 'none')
        .attr('stroke', layer % 2 ? '#e8f5e8' : '#f3e5f5')
        .attr('stroke-width', 0.1)
        .attr('opacity', 0.3 - layer * 0.1);
    }
  }
  
  startAnimation() {
    let time = 0;
    
    const animate = () => {
      time += 0.02;
      
      // Gentle floating animation
      this.svg.select('#angel-body')
        .attr('transform', `translate(0, ${Math.sin(time) * 0.3})`);
      
      // Halo glow pulsing
      this.svg.select('#angel-halo circle')
        .attr('opacity', 0.6 + Math.sin(time * 2) * 0.2);
      
      requestAnimationFrame(animate);
    };
    
    animate();
  }
}

// Initialize when DOM is ready
document.addEventListener('DOMContentLoaded', () => {
  new AngelExplorer();
});
</script>
