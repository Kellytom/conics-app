## ğŸ“ Mathematical Foundation

### Standard Form
<div class="formula-display">
<span class="math-block">y = axÂ² + bx + c</span>
</div>

### Vertex Form  
<div class="formula-display">
<span class="math-block">y = a(x - h)Â² + k</span>
</div>

Where:
- <span class="priority-green">â—</span> **a** determines the width and direction of opening
- <span class="priority-green">â—</span> **h** is the horizontal shift (vertex x-coordinate)  
- <span class="priority-green">â—</span> **k** is the vertical shift (vertex y-coordinate)

<Aside type="note">
**Mathematical Insight**: The coefficient 'a' controls both the width of the parabola and whether it opens upward (a > 0) or downward (a < 0). Larger values of |a| create narrower parabolas.
</Aside>

<script type="module">
// Parabola visualization script
import * as d3 from 'https://cdn.skypack.dev/d3@7';

class ParabolaExplorer {
  constructor() {
    this.svg = d3.select('#parabola-svg');
    this.width = 500;
    this.height = 400;
    this.margin = { top: 20, right: 20, bottom: 20, left: 20 };
    this.xScale = d3.scaleLinear().domain([-10, 10]).range([0, this.width]);
    this.yScale = d3.scaleLinear().domain([-10, 10]).range([this.height, 0]);
    
    this.a = 1;
    this.h = 0;
    this.k = 0;
    
    this.initializeControls();
    this.updateVisualization();
  }
  
  initializeControls() {
    const aSlider = document.getElementById('parabola-a');
    const hSlider = document.getElementById('parabola-h');
    const kSlider = document.getElementById('parabola-k');
    
    aSlider.addEventListener('input', (e) => {
      this.a = parseFloat(e.target.value);
      document.getElementById('parabola-a-value').textContent = this.a.toFixed(1);
      this.updateVisualization();
    });
    
    hSlider.addEventListener('input', (e) => {
      this.h = parseFloat(e.target.value);
      document.getElementById('parabola-h-value').textContent = this.h.toFixed(1);
      this.updateVisualization();
    });
    
    kSlider.addEventListener('input', (e) => {
      this.k = parseFloat(e.target.value);
      document.getElementById('parabola-k-value').textContent = this.k.toFixed(1);
      this.updateVisualization();
    });
  }
  
  updateVisualization() {
    this.drawParabola();
    this.drawLatticePoints();
    this.updateIntersections();
  }
  
  drawParabola() {
    const line = d3.line()
      .x(d => this.xScale(d.x))
      .y(d => this.yScale(d.y))
      .curve(d3.curveBasis);
    
    const data = [];
    for (let x = -10; x <= 10; x += 0.1) {
      const y = this.a * Math.pow(x - this.h, 2) + this.k;
      if (y >= -10 && y <= 10) {
        data.push({ x, y });
      }
    }
    
    this.svg.select('#parabola-path')
      .datum(data)
      .attr('d', line);
  }
  
  drawLatticePoints() {
    const points = [];
    for (let x = -10; x <= 10; x++) {
      for (let y = -10; y <= 10; y++) {
        const parabolaY = this.a * Math.pow(x - this.h, 2) + this.k;
        if (Math.abs(y - parabolaY) < 0.1) {
          points.push({ x, y });
        }
      }
    }
    
    const latticeGroup = this.svg.select('#lattice-points');
    const circles = latticeGroup.selectAll('circle').data(points);
    
    circles.enter()
      .append('circle')
      .merge(circles)
      .attr('cx', d => this.xScale(d.x))
      .attr('cy', d => this.yScale(d.y))
      .attr('r', 3)
      .attr('fill', '#27ae60')
      .attr('stroke', '#2c3e50')
      .attr('stroke-width', 1);
    
    circles.exit().remove();
  }
  
  updateIntersections() {
    const intersections = [];
    for (let x = -10; x <= 10; x++) {
      const y = this.a * Math.pow(x - this.h, 2) + this.k;
      if (Number.isInteger(y) && y >= -10 && y <= 10) {
        intersections.push(`(${x}, ${y})`);
      }
    }
    
    document.getElementById('integer-intersections').textContent = 
      intersections.length > 0 ? intersections.join(', ') : 'No integer intersections in current view';
  }
}

// Initialize when DOM is ready
document.addEventListener('DOMContentLoaded', () => {
  new ParabolaExplorer();
});
</script>
